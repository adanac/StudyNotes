#include<stdio.h>
#include<stdlib.h>

struct list
{
	int data;
	struct list *next;
};

int i;
void output(struct list *head);
void creat(struct list *head, int n)//Í·²å·¨
{
	struct list *p, *q;
	head->next=NULL;
	p=head;
	
	for(i=0; i<n; i++)
	{
		q=(struct list*)malloc( sizeof(struct list) );
		scanf("%d", &q->data);

		q->next = p->next;
		p->next=q;
		p=q;
	}
}

void output(struct list *head)
{
	struct list *p=head->next;
	while(p)
	{
		printf("%d  ", p->data);
		p=p->next;
	}
	puts("\n\n");
}
