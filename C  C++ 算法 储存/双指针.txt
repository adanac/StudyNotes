void sortstring(char (*q)[5][80])
{
	int i, j;
	char temp[80];

	for(i = 0; i < 5; i++)
		for(j = i; j < 5; j++)
			if(strcmp((*q)[i], (*q)[j]) > 0)
			{
				strcpy(temp, (*q)[i]);
				strcpy((*q)[i], (*q)[j]);
				strcpy((*q)[j], temp);
			}
}
