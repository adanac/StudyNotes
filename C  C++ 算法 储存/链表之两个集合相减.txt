#include<stdio.h>
#include<stdlib.h>
typedef struct node
{
	int data;
	struct node * next;
}list;
int i;
list *creat1(int n)
{
	list *head, *p, *q;
	printf("Input %d numbers\n", n);
	for(i=0; i<n; i++)
	{
		p = (list *)malloc( sizeof(list) );
		scanf("%d", &p->data);
		if(i==0)
			head = p;
		else
			q->next = p;	
		q = p;
	}
	q->next = NULL;//尾节点必须置空
	return head;
}
